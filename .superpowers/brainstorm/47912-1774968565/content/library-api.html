token-state 库 API 设计

TypeScript · npm 发布 · 三层使用方式

层 1 · 核心（必须）

TokenTracker — 配额管理

import { TokenTracker } from 'token-state'

// 初始化，指定 state.json 路径
const tracker = new TokenTracker({
  stateFile: '~/.token-hud/state.json' // 默认值
})

// 注册一个 service 及其配额
tracker.addService('claude', {
  label: 'Claude Max',
  quotas: [
    { type: 'time', total: 18000, unit: 'seconds', resetsAt: '2026-04-30' },
    { type: 'tokens', total: 1_000_000 }
  ]
})

// 手动上报消耗
tracker.consume('claude', { tokens: 1500, time: 2.3 })

// 重置配额（如月度续费）
tracker.resetQuota('claude', 'time')
层 2 · Session（推荐）

Session — 追踪单次会话

// 开始一个会话（写入 currentSession）
const session = tracker.startSession('claude')

// 会话中上报消耗（同时更新 quotas.used + currentSession）
session.consume({ tokens: 500, time: 1.2 })

// 结束会话（清空 currentSession → null）
session.end()

// 或用 try/finally 保证 end() 一定调用
await tracker.withSession('claude', async (session) => {
  // ... 你的 AI 调用逻辑 ...
  session.consume({ tokens: 800 })
}) // 自动 end()
层 3 · SDK 插件（可选）

wrapOpenAI / wrapAnthropic — 自动拦截

import { wrapOpenAI } from 'token-state/openai'
import { wrapAnthropic } from 'token-state/anthropic'

// 原来的 client 包一层，其他代码完全不变
const openai = wrapOpenAI(new OpenAI(), tracker, 'openai')
const claude = wrapAnthropic(new Anthropic(), tracker, 'claude')

// 之后正常用，token 自动扣减 ✓
const res = await openai.chat.completions.create({...})
// 内部自动: tracker.consume('openai', { tokens: res.usage.total_tokens })
